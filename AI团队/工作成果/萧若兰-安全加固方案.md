## 1. 凭证安全

### 1.1 当前风险

经审计，eCompany 项目在凭证管理方面存在以下问题：

| 风险点 | 严重程度 | 说明 |
|--------|----------|------|
| API Key 明文存储于配置文件 | 🔴 **高危** | `ai-provider.json` 直接存储 DeepSeek API Key，未加密 |
| 多次冗余存储 | 🟡 **中危** | `DEEPSEEK_API_KEY` 环境变量 + `ai-provider.json` + 运行期内存，存在多个副本 |
| 配置文件写入前未脱敏还原 | 🟡 **中危** | `POST /api/provider/config` 在前端传回 `****` 脱敏值时，`delete body.apiKey` 位于 `saveAIProviderConfig(newCfg)` 之后，导致 `newCfg` 仍用真实值写入，实际未处理前端回传的脱敏 Key |
| License 文件明文存储 | 🟡 **中危** | `licenses.json` 存有激活码，设备绑定信息可被直接读取 |
| 微信 Bot Token 文件 | 🟡 **中危** | `.wx-bot-state.json` 存有长期有效 Bot Token，可被进程外文件读取 |

### 1.2 加固方案

#### 1.2.1 API Key 加密存储

**方案 A（推荐）：OS 级密钥管理 + 环境变量**

```javascript
// backend/modules/security/credential-store.js
const crypto = require('node:crypto');
const fs = require('node:fs');
const path = require('node:path');

const MASTER_KEY_ENV = 'ECOMPANY_MASTER_KEY'; // 系统级主密钥（通过环境变量注入）

/**
 * 加密存储敏感凭证
 * 使用 AES-256-GCM 加密，存储到加密文件
 */
class CredentialStore {
  constructor(storePath) {
    this.storePath = storePath || path.join(__dirname, '../../.encrypted-creds');
    this.masterKey = this._getMasterKey();
  }

  _getMasterKey() {
    // 优先从环境变量读，32字节 hex 格式
    const keyHex = process.env[MASTER_KEY_ENV];
    if (keyHex && keyHex.length === 64) {
      return Buffer.from(keyHex, 'hex');
    }
    // 回退：从受保护的文件读（仅限开发环境）
    const keyFile = path.join(__dirname, '../../.master-key');
    if (fs.existsSync(keyFile)) {
      const mode = fs.statSync(keyFile).mode;
      // 在 Unix 下只允许 owner 读写
      return fs.readFileSync(keyFile, 'utf-8').trim();
    }
    throw new Error(`未设置主密钥。请设置环境变量 ${MASTER_KEY_ENV}=<64位hex> 或创建 .master-key 文件`);
  }

  /** 加密并写入存储 */
  set(key, value) {
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipheriv('aes-256-gcm', this.masterKey, iv);
    let encrypted = cipher.update(JSON.stringify(value), 'utf-8', 'hex');
    encrypted += cipher.final('hex');
    const authTag = cipher.getAuthTag().toString('hex');
    
    let store = {};
    try {
      store = JSON.parse(fs.readFileSync(this.storePath, 'utf-8'));
    } catch (e) { /* 新文件 */ }
    
    store[key] = { iv: iv.toString('hex'), authTag, data: encrypted };
    fs.writeFileSync(this.storePath, JSON.stringify(store, null, 2), { mode: 0o600 });
  }

  /** 解密读取 */
  get(key) {
    let store = {};
    try {
      store = JSON.parse(fs.readFileSync(this.storePath, 'utf-8'));
    } catch (e) { return null; }
    
    const entry = store[key];
    if (!entry) return null;
    
    const decipher = crypto.createDecipheriv(
      'aes-256-gcm',
      this.masterKey,
      Buffer.from(entry.iv, 'hex')
    );
    decipher.setAuthTag(Buffer.from(entry.authTag, 'hex'));
    let decrypted = decipher.update(entry.data, 'hex', 'utf-8');
    decrypted += decipher.final('utf-8');
    return JSON.parse(decrypted);
  }
}

module.exports = CredentialStore;
```

## 2. Web 安全

### 2.1 当前风险

| 风险点 | 严重程度 | 说明 |
|--------|----------|------|
| 完全开放的 CORS | 🟡 **中危** | 所有响应头均返回 `Access-Control-Allow-Origin: *`，无白名单限制 |
| 无认证鉴权 | 🔴 **高危** | 所有 API 端点（员工/任务/记忆/许可/频道消息）均无任何形式认证 |
| 无输入验证 | 🟡 **中危** | `POST /api/memory/sync` 虽有基本路径穿越检查，仍可被绕过 |
| 许可接口无权限控制 | 🟡 **中危** | `POST /api/license/activate` 无需认证即可激活任意许可 |
| 缺少速率限制 | 🟡 **中危** | 无请求频率限制，可被暴力遍历 |
| 无防护头 | 🟢 **低危** | 未设置 CSP、X-Frame-Options、X-Content-Type-Options 等安全头 |
| 敏感信息泄露 | 🟡 **中危** | `/api/provider/config` 返回部分 API Key；错误消息可能泄露内部路径 |

#### 2.2.1 CORS 白名单加固

```javascript
// backend/modules/security/cors.js
const ALLOWED_ORIGINS = [
  'http://localhost:8003',
  'http://localhost:21104',
  'http://127.0.0.1:8003',
  'http://127.0.0.1:21104',
  // 生产环境域名
  'https://ecompany.example.com',
];

function getCorsHeaders(origin) {
  if (!origin || ALLOWED_ORIGINS.includes(origin)) {
    return {
      'Access-Control-Allow-Origin': origin || 'http://localhost:8003',
      'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type, Authorization',
      'Access-Control-Max-Age': '86400',
    };
  }
  // 不信任的来源 → 不返回 CORS 头（浏览器会拒绝）
  return {};
}

module.exports = { getCorsHeaders, ALLOWED_ORIGINS };
```

#### 2.2.3 输入验证中间件

```javascript
// backend/modules/security/validator.js

/**
 * 通用输入验证器
 */
const VALIDATORS = {
  /** 验证任务标题 */
  taskTitle: (val) => {
    if (!val || typeof val !== 'string') return '任务标题不能为空';
    if (val.length > 200) return '任务标题不能超过200字符';
    if (/[<>"']/.test(val)) return '任务标题不能包含 HTML 特殊字符';
    return null; // 通过
  },

  /** 验证消息内容 */
  message: (val) => {
    if (!val || typeof val !== 'string') return '消息内容不能为空';
    if (val.length > 10000) return '消息内容过长';
    return null;
  },

  /** 验证 agent ID */
  agentId: (val, agentsMap) => {
    if (!val || typeof val !== 'string') return '员工ID无效';
    if (!agentsMap[val]) return '员工不存在';
    return null;
  },

  /** 验证任务状态 */
  taskStatus: (val) => {
    const valid = ['todo', 'assigned', 'in_progress', 'done', 'blocked'];
    if (val && !valid.includes(val)) return `状态无效，允许: ${valid.join(', ')}`;
    return null;
  },

  /** 验证优先级 */
  priority: (val) => {
    const valid = ['low', 'medium', 'high', 'critical'];
    if (val && !valid.includes(val)) return `优先级无效，允许: ${valid.join(', ')}`;
    return null;
  },

  /** 验证文件路径（防路径穿越） */
  filename: (val) => {
    if (!val || typeof val !== 'string') return '文件名无效';
    if (val.includes('..') || val.includes('/') || val.includes('\\')) return '文件名不允许包含路径分隔符';
    if (!/^[a-zA-Z0-9._-]+$/.test(val)) return '文件名只允许字母、数字、点、下划线和连字符';
    if (val.length > 255) return '文件名过长';
    return null;
  },
};

/**
 * 批量验证请求体
 * @param {object} body - 请求体
 * @param {object} rules - { fieldName: validatorFn }
 * @returns {string|null} 错误信息或 null
 */
function validate(body, rules) {
  for (const [field, validator] of Object.entries(rules)) {
    const err = validator(body[field]);
    if (err) return err;
  }
  return null;
}

module.exports = { VALIDATORS, validate };
```

#### 2.2.4 认证鉴权方案

由于 eCompany 当前是一个本地/内网团队协作平台，建议分阶段实施：

**阶段一：简单 Token 认证（快速落地）**

```javascript
// backend/modules/security/auth.js
const crypto = require('node:crypto');

// 从环境变量或配置读取
// 生成方式: node -e "console.log(crypto.randomBytes(32).toString('hex'))"
const ACCESS_TOKEN = process.env.ECOMPANY_ACCESS_TOKEN || ''; 

function authenticate(req, res) {
  // 本地环回地址免认证（方便本地开发）
  if (req.socket && (req.socket.remoteAddress === '127.0.0.1' || req.socket.remoteAddress === '::1')) {
    return true;
  }
  
  if (!ACCESS_TOKEN) return true; // 未配置 Token 时放行（兼容旧版）
  
  const authHeader = req.headers['authorization'] || '';
  const token = authHeader.startsWith('Bearer ') ? authHeader.slice(7) : '';
  
  if (token !== ACCESS_TOKEN) {
    res.writeHead(401, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify({ error: '未授权，请提供有效的 Bearer Token' }));
    return false;
  }
  return true;
}

module.exports = { authenticate, ACCESS_TOKEN };
```

**阶段二：Web 登录界面 + Session**

对于给前端添加登录，建议方案：

1. **管理员账号**：单页登录界面，硬编码/配置化 admin 密码
2. **JWT Session**：登录后签发短期 JWT Token
3. **前端拦截器**：401 时自动跳转登录页

```javascript
// backend/modules/security/session-auth.js
const crypto = require('node:crypto');

// 管理员凭据（从加密文件或环境变量读取）
const ADMIN_HASH = process.env.ECOMPANY_ADMIN_HASH || '';
// node -e "console.log(crypto.createHash('sha256').update('your-password').digest('hex'))"

// 活跃 Session（简单内存管理，生产环境建议 Redis）
const sessions = new Map();
const SESSION_TTL = 8 * 60 * 60 * 1000; // 8 小时

function createSession(username) {
  const sessionId = crypto.randomUUID();
  sessions.set(sessionId, {
    username,
    createdAt: Date.now(),
    expiresAt: Date.now() + SESSION_TTL,
  });
  return sessionId;
}

function validateSession(sessionId) {
  const session = sessions.get(sessionId);
  if (!session) return null;
  if (Date.now() > session.expiresAt) {
    sessions.delete(sessionId);
    return null;
  }
  return session;
}

function revokeSession(sessionId) {
  sessions.delete(sessionId);
}

// 登录验证
function verifyPassword(password) {
  if (!ADMIN_HASH) return true; // 未配置密码时放行
  const hash = crypto.createHash('sha256').update(password).digest('hex');
  return hash === ADMIN_HASH;
}

module.exports = { createSession, validateSession, revokeSession, verifyPassword };
```

```javascript
// backend/modules/security/log-sanitizer.js

/**
 * 敏感模式匹配（用于日志脱敏）
 */
const SENSITIVE_PATTERNS = [
  // API Keys
  { pattern: /(sk-[a-zA-Z0-9]{20,})/g, replacement: 'sk-****' },
  { pattern: /(apiKey['"]?\s*[:=]\s*['"])([^'"]{4})([^'"]+)(['"])/g, replacement: '$1$2****$4' },
  // 令牌
  { pattern: /(token['"]?\s*[:=]\s*['"])([^'"]{4})([^'"]+)(['"])/g, replacement: '$1$2****$4' },
  // Bearer Token
  { pattern: /(Bearer\s+)([a-zA-Z0-9_-]{8,})(\s|$)/g, replacement: '$1****' },
  // 密码
  { pattern: /(password['"]?\s*[:=]\s*['"])([^'"]+)(['"])/g, replacement: '$1****$3' },
  // 手机号
  { pattern: /(1[3-9]\d)\d{4}(\d{4})/g, replacement: '$1****$2' },
  // 微信 Bot Token（变长）
  { pattern: /(token["']?\s*[:=]\s*["'])([^"']{4})([^"']+)(["'])/g, replacement: '$1$2****$4' },
];

/**
 * 脱敏日志消息
 */
function sanitizeLog(str) {
  if (typeof str !== 'string') return str;
  let result = str;
  for (const { pattern, replacement } of SENSITIVE_PATTERNS) {
    result = result.replace(pattern, replacement);
  }
  return result;
}

/**
 * 安全日志输出（替代 console.log）
 */
const LOG_LEVELS = { DEBUG: 0, INFO: 1, WARN: 2, ERROR: 3 };
let currentLogLevel = process.env.LOG_LEVEL || 'INFO';

function safeLog(level, module, message, data) {
  if (LOG_LEVELS[level] < LOG_LEVELS[currentLogLevel]) return;
  
  const entry = {
    timestamp: new Date().toISOString(),
    level,
    module,
    message: sanitizeLog(typeof message === 'string' ? message : JSON.stringify(message)),
  };
  
  if (data) {
    entry.data = sanitizeLog(typeof data === 'string' ? data : JSON.stringify(data));
  }
  
  const line = JSON.stringify(entry);
  
  switch (level) {
    case 'ERROR': console.error(line); break;
    case 'WARN':  console.warn(line);  break;
    default:      console.log(line);   break;
  }
  
  // 同时写入文件日志
  try {
    const logDir = path.join(BASE, 'logs');
    if (!fs.existsSync(logDir)) fs.mkdirSync(logDir, { recursive: true });
    fs.appendFileSync(path.join(logDir, 'app.secure.log'), entry.message + '\n', 'utf-8');
  } catch (e) { /* 日志写入失败不做处理 */ }
}

module.exports = { sanitizeLog, safeLog };
```

```javascript
// backend/modules/security/rate-limiter.js
/**
 * 简单内存限速器（生产环境建议使用 Redis）
 */
class RateLimiter {
  constructor(options = {}) {
    this.windowMs = options.windowMs || 60000; // 窗口（毫秒）
    this.maxAttempts = options.maxAttempts || 100; // 窗口内最大请求数
    this.store = new Map();
  }

  /** 检查 IP 是否允许请求 */
  check(ip) {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    
    if (!this.store.has(ip)) {
      this.store.set(ip, [now]);
      return { allowed: true };
    }
    
    const attempts = this.store.get(ip).filter(t => t > windowStart);
    attempts.push(now);
    this.store.set(ip, attempts);
    
    return {
      allowed: attempts.length <= this.maxAttempts,
      remaining: Math.max(0, this.maxAttempts - attempts.length),
      resetAt: new Date(now + this.windowMs).toISOString(),
    };
  }
}

// 敏感端点使用更严格限制
const loginLimiter = new RateLimiter({ windowMs: 300000, maxAttempts: 5 }); // 5 分钟 5 次
const apiLimiter = new RateLimiter({ windowMs: 60000, maxAttempts: 100 });  // 1 分钟 100 次

module.exports = { RateLimiter, loginLimiter, apiLimiter };
```

```javascript
class ProcessSandbox {
  constructor(options = {}) {
    this.timeout = options.timeout || 10000;  // 降低默认超时
    this.maxMemory = options.maxMemory || 64; // 降低默认内存限制
    // ...
    
    // 黑名单命令（禁止执行的 shell 命令）
    this.blockedCommands = [
      'rm -rf', 'del', 'format', 'shutdown', 'reboot',
      'sudo', 'su', 'chmod 777', 'chown',
      'curl', 'wget', 'nc', 'ncat', 'telnet',
      'eval', 'exec', 'spawn', 'fork', '`',
      'fs.writeFileSync', 'fs.writeFile', 'fs.appendFile',
      'require(\'child_process\')', 'process.binding',
    ];
  }
  
  async execute(code, language = 'js', options = {}) {
    // 代码安全扫描
    const scanResult = this._securityScan(code, language);
    if (!scanResult.safe) {
      throw new Error(`沙箱安全拦截: ${scanResult.reason}`);
    }
    // ... 原有逻辑
  }
  
  _securityScan(code, language) {
    if (language === 'js') {
      // 禁止 require 文件系统/网络操作
      if (/require\s*\(\s*['"]fs['"]/.test(code) && !/allow-fs/.test(options)) {
        return { safe: false, reason: '不允许直接访问文件系统' };
      }
      if (/child_process/.test(code)) {
        return { safe: false, reason: '不允许执行子进程' };
      }
      if (/process\.(env|binding|chdir|kill|exit)/.test(code)) {
        return { safe: false, reason: '不允许访问系统进程' };
      }
    }
    return { safe: true };
  }
}
```

```markdown
- 紧急情况直接联系方浩铭（安全总监）

**2. 事件分类**
- P0 (立即响应): 数据泄露、系统被入侵、凭证泄露
- P1 (4小时内): 漏洞被利用、权限提升

**3. 响应步骤**
1. 确认 → 2. 隔离 → 3. 取证 → 4. 清除 → 5. 恢复 → 6. 复盘

**4. 事后复盘**
- 根因分析
- 措施跟进
- 文档更新
- 全员通告（如适用）
```

### A. 快速安全检测脚本

```bash
#!/bin/bash
# ecompany-security-check.sh

echo "=== eCompany 安全检查 ==="

# 1. 检查敏感文件是否明文存储密钥
echo ""
echo "[1/7] 检查明文密钥存储..."
if grep -q "apiKey" "backend/ai-provider.json" 2>/dev/null; then
  echo "  ⚠️  ai-provider.json 可能包含 API Key"
else
  echo "  ✅ ai-provider.json 未包含 API Key"
fi

# 2. 检查 .env 文件是否存在（不应提交到 Git）
echo ""
echo "[2/7] 检查 .env 文件..."
if [ -f ".env" ] && grep -q "DEEPSEEK" ".env"; then
  echo "  ⚠️  .env 文件存在且包含 API Key，确认已加入 .gitignore"
else
  echo "  ✅ .env 文件状态正常"
fi

# 3. 检查 npm audit
echo ""
echo "[3/7] 依赖漏洞扫描 (npm audit)..."
cd backend && npm audit --audit-level=high 2>/dev/null && echo "  ✅ 无高危漏洞" || echo "  ⚠️  存在高危漏洞，请运行 npm audit fix"

# 4. 检查安全头
echo ""
echo "[4/7] HTTP 安全头检查..."
curl -s -D- http://localhost:8003/api/health 2>/dev/null | head -20 | grep -i "x-content-type-options\|x-frame-options\|content-security-policy" > /dev/null && echo "  ✅ 安全头配置正常" || echo "  ⚠️  安全头未正确配置"

# 5. 检查 CORS
echo ""
echo "[5/7] CORS 配置检查..."
curl -s -D- -H "Origin: https://evil.com" http://localhost:8003/api/health 2>/dev/null | grep -i "access-control-allow-origin" | grep -v "localhost" > /dev/null && echo "  ⚠️  CORS 可能配置为通配符 *" || echo "  ✅ CORS 白名单有效"

# 6. 检查日志脱敏
echo ""
echo "[6/7] 日志脱敏检查..."
if grep -q "sk-" logs/app.log 2>/dev/null; then
  echo "  ⚠️  日志可能包含 API Key"
else
  echo "  ✅ 日志未发现明文密钥"
fi

# 7. 检查文件权限
echo ""
echo "[7/7] 敏感文件权限检查..."
if [ -f "backend/.encrypted-creds" ]; then
  perms=$(stat -c "%a" backend/.encrypted-creds 2>/dev/null)
  echo "  .encrypted-creds 权限: $perms（应为 600）"
fi

echo ""
echo "=== 安全检查完成 ==="
```
